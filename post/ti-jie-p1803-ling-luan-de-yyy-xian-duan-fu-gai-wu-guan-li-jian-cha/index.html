题解 P1803 【凌乱的yyy / 线段覆盖】 无管理检查

 2022 年 01 月 23日

这道题照样看了很多大佬的题解。

其实，这篇说是题解，其实相当于将另一位大佬的题解换一种写法。

大体如下：

#include <bits/stdc++.h>
using namespace std;
struct oj{ int x,y; }a[1000000];
bool vis[1000000];
inline int read_int(){
	int res=0,f=1;
	char ch=getchar();
	while(ch<'0'||'9'<ch){
		if(ch=='-') f=-1;
		ch=getchar();
	}
	while('0'<=ch&&ch<='9')
		res=res*10+(ch-'0'),ch=getchar();
	return res*f;
}
bool cmp(oj a,oj b){
	return a.y<b.y;
}
int n,ans;
int main(){
	memset(vis,0,sizeof vis);
	cin>>n;
	for(int i=1;i<=n;i++)
		a[i]={read_int(),read_int()};
	sort(a+1,a+n+1,cmp);
	for(int i=1;i<=n;i++)
		if(!vis[i]){
			ans++,vis[i]=!vis[i];
			for(int j=i+1;a[i].y>a[j].x;j++) vis[j]=true;
		}
	cout<<ans;
	return 0;
}
